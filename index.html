###REQUEST THE BLS PAGE ON SIMILAR OCCUPATIONS TO STATISTICIANS
#Request and make some sexy soup out of the url.
res = requests.get("https://www.bls.gov/ooh/math/statisticians.htm#tab-8")
soup = BeautifulSoup(res.content,'lxml') #Get all of the 'lxml' content from the page

#There are multiple tables on the page. Decided the 5th one was of interest
table = soup.find_all('table')[4] #Find all tables, assign the 5th one to a table object
df = pd.read_html(str(table))[0] #Turn that soup into a str, then read the html to a dataframe
df #Let's see that dataframe boys!

"""Clean up the table."""
clean_table = df[["Occupation", "Job Duties", "ENTRY-LEVEL EDUCATION", "2016 MEDIAN PAY"]]

#Clean up the median pay with re and a generator comprehension that removes '$' and ',' from "2016 MEDIAN PAY"]
clean_table["MP_2016_Clean"] = [re.sub('[$,]', '', pay) for pay in clean_table["2016 MEDIAN PAY"]]
clean_table["MP_2016_Clean"] = [int(pay) for pay in clean_table["MP_2016_Clean"]]

print("Now she's a little more analyzable.")
clean_table[["2016 MEDIAN PAY", "MP_2016_Clean"]]

y_pos = np.arange(len(clean_table["Occupation"]))
pay = clean_table["MP_2016_Clean"]

plt.barh(y_pos, pay, align = 'center', alpha = 0.5)
plt.yticks(y_pos, occ)

plt.title("2016 Median Pay for Occupations Related to Statisticians")
